#%% Q2 setup, 60pts
import scipy.stats as stats
import numpy.random as rn
#%% Q2a, 15pts
rn.seed(0)
N = 16;
m = 10**5;
# 10**5 by 16 = 16 + 16 + ... + 16
# use the uniform or binomial functions to generate the data
nHeads = (rn.uniform(size=(m, N))>0.5).sum(1)

#%%plot the histgram
# Your code is here
# suggest one line code

count, bins, bars = plt.hist(nHeads, bins=range(18))

plt.xlabel('nHeads')
plt.ylabel('Number of coins')
plt.title('Q2a: Histogram')
plt.show()

#%% Q2b, 10pts
# using plt.hist with parameter bins and density
#Your code is here

plt.hist(nHeads, bins=range(18), density=True)

plt.xlabel('k')
plt.ylabel('P(nHeads = k)')
plt.title('Q2b: Probability Mass Function')
plt.show()

#%% Q2c, 15pts

#Use the values of counts returned from a or b. 
#Use the function cumsum to get the CDF.
# Do not use scipy.stats.binom
# Your code is here

# print(np.cumsum(nHeads))
pdf = count / sum(count)
cdf = np.cumsum(pdf)
plt.plot(bins[1:], cdf)

plt.xlabel('k')
plt.ylabel('P(nHeads <= k)')
plt.title('Q2c: Cumulative Distribution Function')
plt.show()
